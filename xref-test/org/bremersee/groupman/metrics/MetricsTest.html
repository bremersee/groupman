73  @ActiveProfiles({"basic-auth", "ldap"})
74  @TestInstance(Lifecycle.PER_CLASS) // allows us to use @BeforeAll with a non-static method
75  public class MetricsTest {
76  
77    private static final GroupEntity group0 = GroupEntity.builder()
78        .id("MT0")
79        .source(Source.INTERNAL)
80        .name("Group0")
81        .description("Group One")
82        .createdAt(new Date())
83        .modifiedAt(new Date())
84        .createdBy("molly")
85        .owners(Collections.singleton("molly"))
86        .members(Stream.of("molly", "leopold").collect(Collectors.toSet()))
87        .build();
88  
89    private static final GroupEntity group1 = GroupEntity.builder()
90        .id("MT1")
91        .source(Source.INTERNAL)
92        .name("Group1")
93        .description("Group Two")
94        .createdAt(new Date())
95        .modifiedAt(new Date())
96        .createdBy("molly")
97        .owners(Collections.singleton("molly"))
98        .build();
99  
100   private static final GroupEntity group2 = GroupEntity.builder()
101       .id("MT2")
102       .source(Source.INTERNAL)
103       .name("Group2")
104       .description("Group Three")
105       .createdAt(new Date())
106       .modifiedAt(new Date())
107       .createdBy("leopold")
108       .owners(Stream.of("leopold").collect(Collectors.toSet()))
109       .members(Stream.of("molly", "leopold", "stephen").collect(Collectors.toSet()))
110       .build();
111 
112   /**
113    * The application context.
114    */
115   @Autowired
116   ApplicationContext context;
117 
118   /**
119    * The web test client.
120    */
121   @SuppressWarnings("SpringJavaInjectionPointsAutowiringInspection")
122   @Autowired
123   WebTestClient webTestClient;
124 
125   /**
126    * The Group repository.
127    */
128   @Autowired
129   GroupRepository groupRepository;
130 
131   /**
132    * Setup tests.
133    */
134   @BeforeAll
135   void setUp() {
136     // https://docs.spring.io/spring-security/site/docs/current/reference/html/test-webflux.html
137     WebTestClient
138         .bindToApplicationContext(this.context)
139         .configureClient()
140         .build();
141   }
142 
143   /**
144    * Sets up data.
145    */
146   @BeforeAll
147   void setUpData() {
148     StepVerifier
149         .create(groupRepository.save(group0))
150         .consumeNextWith(groupEntity -> {
151           assertNotNull(groupEntity.getId());
152           assertEquals("MT0", groupEntity.getId());
153         })
154         .verifyComplete();
155     StepVerifier
156         .create(groupRepository.save(group1))
157         .consumeNextWith(groupEntity -> {
158           assertNotNull(groupEntity.getId());
159           assertEquals("MT1", groupEntity.getId());
160         })
161         .verifyComplete();
162     StepVerifier
163         .create(groupRepository.save(group2))
164         .consumeNextWith(groupEntity -> {
165           assertNotNull(groupEntity.getId());
166           assertEquals("MT2", groupEntity.getId());
167         })
168         .verifyComplete();
169   }
170 
171   /**
172    * Call prometheus endpoint.
173    */
174   @WithMockUser(
175       username = "actuator",
176       password = "actuator",
177       authorities = {ACTUATOR_ROLE_NAME, ACTUATOR_ADMIN_ROLE_NAME})
178   @Test
179   void callPrometheusEndpoint() {
180     webTestClient
181         .get()
182         .uri("/actuator/prometheus")
183         .exchange()
184         .expectStatus().isOk()
185         .expectBody(String.class)
186         .value(body -> {
187           System.out.println("Prometheus data:\n" + body);
188           assertTrue(body.contains("groups_size{storage=\"ldap\",} 2.0"));
189           assertTrue(body.contains("groups_size{storage=\"mongodb\",} 3.0"));
190         });
191   }
