17  package org.bremersee.groupman.config;
18  
19  import lombok.extern.slf4j.Slf4j;
20  import org.bremersee.security.authentication.KeycloakReactiveJwtConverter;
21  import org.bremersee.security.authentication.PasswordFlowReactiveAuthenticationManager;
22  import org.bremersee.security.core.AuthorityConstants;
23  import org.springframework.beans.factory.annotation.Autowired;
24  import org.springframework.boot.actuate.autoconfigure.security.reactive.EndpointRequest;
25  import org.springframework.boot.actuate.health.HealthEndpoint;
26  import org.springframework.boot.actuate.info.InfoEndpoint;
27  import org.springframework.context.annotation.Bean;
28  import org.springframework.context.annotation.Configuration;
29  import org.springframework.core.annotation.Order;
30  import org.springframework.security.config.annotation.web.reactive.EnableWebFluxSecurity;
31  import org.springframework.security.config.web.server.ServerHttpSecurity;
32  import org.springframework.security.web.server.SecurityWebFilterChain;
33  import org.springframework.security.web.server.util.matcher.NegatedServerWebExchangeMatcher;
34  
35  /**
36   * The security configuration.
37   *
38   * @author Christian Bremer
39   */
40  @Configuration
41  @EnableWebFluxSecurity
42  @Slf4j
43  public class SecurityConfiguration {
44  
45    private KeycloakReactiveJwtConverter keycloakJwtConverter;
46  
47    private PasswordFlowReactiveAuthenticationManager passwordFlowReactiveAuthenticationManager;
48  
49    /**
50     * Instantiates a new security configuration.
51     *
52     * @param keycloakJwtConverter                      the keycloak jwt converter
53     * @param passwordFlowReactiveAuthenticationManager the password flow reactive authentication
54     *                                                  manager
55     */
56    @Autowired
57    public SecurityConfiguration(
58        KeycloakReactiveJwtConverter keycloakJwtConverter,
59        PasswordFlowReactiveAuthenticationManager passwordFlowReactiveAuthenticationManager) {
60      this.keycloakJwtConverter = keycloakJwtConverter;
61      this.passwordFlowReactiveAuthenticationManager = passwordFlowReactiveAuthenticationManager;
62    }
63  
64    /**
65     * Builds the OAuth2 resource server filter chain.
66     *
67     * @param http the http
68     * @return the security web filter chain
69     */
70    @Bean
71    @Order(51)
72    public SecurityWebFilterChain oauth2ResourceServerFilterChain(ServerHttpSecurity http) {
73  
74      log.info("msg=[Creating resource server filter chain.]");
75      http
76          .securityMatcher(new NegatedServerWebExchangeMatcher(EndpointRequest.toAnyEndpoint()))
77          .csrf().disable()
78          .oauth2ResourceServer()
79          .jwt()
80          .jwtAuthenticationConverter(keycloakJwtConverter);
81  
82      http
83          .authorizeExchange()
84          .pathMatchers("/api/admin/**")
85          .hasAuthority(AuthorityConstants.ADMIN_ROLE_NAME)
86          .anyExchange()
87          .authenticated();
88  
89      return http.build();
90    }
91  
92    /**
93     * Builds the actuator filter chain.
94     *
95     * @param http the http security configuration object
96     * @return the security web filter chain
97     */
98    @Bean
99    @Order(52)
100   public SecurityWebFilterChain actuatorFilterChain(ServerHttpSecurity http) {
101 
102     log.info("msg=[Creating actuator filter chain.]");
103     http
104         .securityMatcher(EndpointRequest.toAnyEndpoint())
105         .csrf().disable()
106         .httpBasic()
107         .authenticationManager(passwordFlowReactiveAuthenticationManager);
108 
109     http
110         .authorizeExchange()
111         .matchers(EndpointRequest.to(HealthEndpoint.class)).permitAll()
112         .matchers(EndpointRequest.to(InfoEndpoint.class)).permitAll()
113         .anyExchange().hasAuthority(AuthorityConstants.ACTUATOR_ROLE_NAME);
114 
115     return http.build();
116   }
117 
118 }
