54  /**
55   * The abstract group controller.
56   *
57   * @author Christian Bremer
58   */
59  abstract class AbstractGroupController {
60  
61    /**
62     * The default sort order.
63     */
64    static final Sort SORT = Sort.by("name", "createdBy");
65  
66    /**
67     * The default comparator.
68     */
69    static final Comparator<Object> COMPARATOR = ComparatorBuilder.builder()
70        .addAll(ComparatorSpringUtils.fromSort(SORT))
71        .build();
72  
73    @Getter(AccessLevel.PACKAGE)
74    private final ModelMapper modelMapper = new ModelMapper();
75  
76    @Getter(AccessLevel.PACKAGE)
77    private GroupRepository groupRepository;
78  
79    @Getter(AccessLevel.PACKAGE)
80    private GroupLdapRepository groupLdapRepository;
81  
82    private String localUserRole;
83  
84    /**
85     * Instantiates a new abstract group controller.
86     *
87     * @param groupRepository     the group repository
88     * @param groupLdapRepository the group ldap repository
89     * @param localUserRole       the local user role
90     */
91    public AbstractGroupController(
92        final GroupRepository groupRepository,
93        final GroupLdapRepository groupLdapRepository,
94        final String localUserRole) {
95  
96      Assert.notNull(groupRepository, "Group repository must not be null.");
97      Assert.notNull(groupLdapRepository, "Group ldap repository must not be null.");
98      this.groupRepository = groupRepository;
99      this.groupLdapRepository = groupLdapRepository;
100     this.localUserRole = localUserRole;
101     modelMapper.addConverter(new AbstractConverter<Date, OffsetDateTime>() {
102       @Override
103       protected OffsetDateTime convert(Date date) {
104         return date == null ? null : OffsetDateTime.ofInstant(date.toInstant(), ZoneId.of("UTC"));
105       }
106     });
107     modelMapper.addConverter(new AbstractConverter<OffsetDateTime, Date>() {
108       @Override
109       protected Date convert(OffsetDateTime offsetDateTime) {
110         return offsetDateTime == null ? null : Date.from(offsetDateTime.toInstant());
111       }
112     });
113   }
114 
115   /**
116    * One with current user mono.
117    *
118    * @param <R>      the type parameter
119    * @param function the function
120    * @return the mono
121    */
122   <R> Mono<R> oneWithCurrentUser(Function<CurrentUser, ? extends Mono<R>> function) {
123     return ReactiveSecurityContextHolder.getContext()
124         .map(SecurityContext::getAuthentication)
125         .switchIfEmpty(Mono.error(ServiceException::forbidden))
126         .map(authentication -> new CurrentUser(authentication, localUserRole))
127         .flatMap(function);
128   }
129 
130   /**
131    * Many with current user flux.
132    *
133    * @param <R>      the type parameter
134    * @param function the function
135    * @return the flux
136    */
137   <R> Flux<R> manyWithCurrentUser(Function<CurrentUser, ? extends Publisher<R>> function) {
138     return ReactiveSecurityContextHolder.getContext()
139         .map(SecurityContext::getAuthentication)
140         .switchIfEmpty(Mono.error(ServiceException::forbidden))
141         .map(authentication -> new CurrentUser(authentication, localUserRole))
142         .flatMapMany(function);
143   }
144 
145   /**
146    * Gets group entity by id.
147    *
148    * @param groupId the group id
149    * @return the group entity
150    */
151   Mono<GroupEntity> getGroupEntityById(final String groupId) {
152     return groupRepository
153         .findById(groupId)
154         .switchIfEmpty(groupLdapRepository.findByName(groupId))
155         .switchIfEmpty(Mono.error(() -> ServiceException.notFound("Group", groupId)));
156   }
157 
158   /**
159    * Gets group entities by ids.
160    *
161    * @param ids the ids
162    * @return the group entities by ids
163    */
164   Flux<GroupEntity> getGroupEntitiesByIds(final List<String> ids) {
165     return groupRepository
166         .findByIdIn(ids == null ? Collections.emptyList() : ids)
167         .concatWith(groupLdapRepository.findByNameIn(ids))
168         .sort(COMPARATOR);
169   }
170 
171   private Group prepareGroup(final Supplier<Group> groupSupplier) {
172     Group group = groupSupplier.get();
173     if (group.getMembers() == null) {
174       group.setMembers(Collections.emptyList());
175     }
176     if (group.getOwners() == null) {
177       group.setOwners(Collections.emptyList());
178     }
179     return group;
180   }
181 
182   /**
183    * Map group entity to group representation.
184    *
185    * @param source the group entity
186    * @return the group representation
187    */
188   Group mapToGroup(final GroupEntity source) {
189     Group destination = new Group();
190     destination.setMembers(new ArrayList<>());
191     destination.setOwners(new ArrayList<>());
192     getModelMapper().map(source, destination);
193     return destination;
194   }
195 
196   /**
197    * Map to group representation to group entity.
198    *
199    * @param source the group representation
200    * @return the group entity
201    */
202   GroupEntity mapToGroupEntity(final Group source) {
203     GroupEntity destination = new GroupEntity();
204     getModelMapper().map(prepareGroup(() -> source), destination);
205     return destination;
206   }
207 
208   /**
209    * Update group entity with group representation.
210    *
211    * @param source              the group representation
212    * @param destinationSupplier the group entity supplier
213    * @return the group entity
214    */
215   GroupEntity updateGroup(final Group source, final Supplier<GroupEntity> destinationSupplier) {
216     final Group src = prepareGroup(() -> source);
217     final GroupEntity destination = destinationSupplier.get();
218     if (source.getVersion() != null) {
219       destination.setVersion(source.getVersion());
220     }
221     destination.setModifiedAt(new Date());
222     destination.setDescription(src.getDescription());
223     destination.setMembers(new LinkedHashSet<>(src.getMembers()));
224     destination.setName(src.getName());
225     destination.setOwners(new LinkedHashSet<>(src.getOwners()));
226     return destination;
227   }
228 
229   /**
230    * The type Current user.
231    */
232   @Getter
233   @ToString
234   @EqualsAndHashCode(of = "name")
235   public static class CurrentUser {
236 
237     private String name;
238 
239     private Set<String> roles;
240 
241     private boolean localUser;
242 
243     /**
244      * Instantiates a new Current user.
245      *
246      * @param authentication the authentication
247      * @param localUserRole  the local user role
248      */
249     public CurrentUser(Authentication authentication, String localUserRole) {
250       name = authentication.getName();
251       if (authentication.getAuthorities() != null) {
252         roles = authentication.getAuthorities().stream()
253             .map(GrantedAuthority::getAuthority)
254             .collect(Collectors.toSet());
255       } else {
256         roles = Collections.emptySet();
257       }
258       localUser = localUserRole != null && roles.contains(localUserRole);
259     }
260   }
